step1：连接
终端1
select * from goods_cates;
step2：增加数据
终端2：开启事务，插入数据
begin;
insert into goods_cates(name) values('小霸王游戏机');
终端2：查询数据，此时有新增的数据
select * from goods_cates;
step3：查询
终端1：查询数据，发现并没有新增的数据
select * from goods_cates;
step4：回滚
终端2：完成回滚
rollback;
step5：查询
终端1：查询数据，发现没有新增的数据
select * from goods_cates;
